cmake_minimum_required(VERSION 2.8)
project(common_vme)
option(UNIVERSE2 "open source universe2 driver" ON)
IF (DEFINED ENV{CHAOS_BUNDLE} )
  include($ENV{CHAOS_BUNDLE}/tools/project_template/CMakeChaos.txt)
ELSE()
  MESSAGE(STATUS "STANDALONE, environment CHAOS_BUNDLE defined")
  IF (NOT CMAKE_INSTALL_PREFIX) 
    MESSAGE(fatal "you must add -DCMAKE_INSTALL_PREFIX=<dir>")
  ELSE()
    MESSAGE(STATUS "install prefix ${CMAKE_INSTALL_PREFIX}")
  ENDIF()
  INCLUDE_DIRECTORIES(. ../.. ${CMAKE_INSTALL_PREFIX}/include)
  LINK_DIRECTORIES(${CMAKE_BINARY_DIR}/build ${CMAKE_INSTALL_PREFIX}/lib)
ENDIF()
ADD_DEFINITIONS(-fpermissive)
IF(DEFINED UNIVERSE2)
ADD_DEFINITIONS(-DUNIVERSE2 -DDEBUG)  
ADD_SUBDIRECTORY(vmedrv/universe2)
ENDIF()

INCLUDE_DIRECTORIES(. ../.. ../ vmedrv vmewrap caen/src sis/)
LINK_DIRECTORIES(lib)
  
SET(sis_src sis/sis3800_drv.c)

SET(vmewrap_src vmewrap/vmewrap.cpp)
SET(caen_src caen/caen965_drv.c caen/caen513_drv.c caen/caen792_drv.c )
SET(caen960_test_src test/test_qdc_caen.c )
SET(test_btf_daq_src test/test_btf_daq.c )



ADD_LIBRARY(${PROJECT_NAME} ${caen_src} ${vmewrap_src} ${sis_src})
ADD_EXECUTABLE(caen_qdc_test ${caen960_test_src})
ADD_EXECUTABLE(test_btf_daq ${test_btf_daq_src})
ADD_EXECUTABLE(test_btf_daq2 test/test_btf_daq2.c)
ADD_EXECUTABLE(test_pio test/test_pio.c)
ADD_EXECUTABLE(test_sis test/test_sis.c)
ADD_EXECUTABLE(vme_read test/vme_read.c)
ADD_EXECUTABLE(vme_write test/vme_write.c)

TARGET_LINK_LIBRARIES(caen_qdc_test ${PROJECT_NAME} vme ${FrameworkLib} common_debug)

TARGET_LINK_LIBRARIES(test_btf_daq2 ${PROJECT_NAME} vme ${FrameworkLib} common_debug)
TARGET_LINK_LIBRARIES(test_pio ${PROJECT_NAME} vme ${FrameworkLib} common_debug)
TARGET_LINK_LIBRARIES(test_sis ${PROJECT_NAME} vme ${FrameworkLib} common_debug)
TARGET_LINK_LIBRARIES(vme_read ${PROJECT_NAME} vme ${FrameworkLib} common_debug)
TARGET_LINK_LIBRARIES(vme_write ${PROJECT_NAME} vme ${FrameworkLib} common_debug)

INSTALL_TARGETS(/lib ${PROJECT_NAME})
INSTALL_TARGETS(/bin caen_qdc_test)
INSTALL_TARGETS(/bin test_btf_daq)

INSTALL_TARGETS(/bin vme_read)

INSTALL(FILES ${src} DESTINATION include/common/vme/vmewrap)
